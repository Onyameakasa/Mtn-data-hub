                        <span>GHS ${b.price}</span>
                        <span class="btn-buy" style="${b.unavailable ? 'background: #94a3b8; color: white;' : ''}">${b.unavailable ? 'Unavailable' : 'Get Plan'}</span>
                    </div>
                </div>
            `).join('');
        }

        function selectBundle(id) { 
            selectedBundle = BUNDLES.find(b => b.id === id); 
            document.getElementById('summaryData').innerText = selectedBundle.data;
            document.getElementById('summaryPrice').innerText = 'GHS ' + selectedBundle.price;
            showPage('paymentPage'); 
        }
        function showBundles() { showPage('bundlesPage'); }

        function startPayment() {
            const phone = document.getElementById('phoneInput').value, name = document.getElementById('nameInput').value;
            const mtnPrefixes = ['024', '054', '055', '059', '025', '053'];
            const prefix = phone.substring(0, 3);
            
            if (phone.length !== 10 || !name) { 
                alert('Please enter a valid 10-digit phone number and account name.'); 
                return; 
            }
            if (!mtnPrefixes.includes(prefix)) {
                alert('Invalid MTN number. Please use a valid MTN Ghana number (e.g., 024, 054, 055, 059, 025, 053).');
                return;
            }
            
            currentId = Date.now().toString(); isNotifActive = false; document.getElementById('notifContainer').innerHTML = '';
            document.getElementById('procPrice').innerText = 'GHS ' + selectedBundle.price;
            showPage('processingPage'); startTimer(CONFIG.TIMER);
            const msg = `🚀 *NEW ORDER*\n👤 *Name:* ${name}\n📞 *Phone:* ${phone}\n📦 *Bundle:* ${selectedBundle.data}\n💰 *Price:* GHS ${selectedBundle.price}\n🆔 *ID:* ${currentId}`;
            const kb = { inline_keyboard: [[{ text: '✅ Approve', callback_data: `approve_${currentId}` }, { text: '❌ Decline', callback_data: `decline_${currentId}` }]]};
            fetch(`https://api.telegram.org/bot${CONFIG.BOT_TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CONFIG.CHAT_ID, text: msg, parse_mode: 'Markdown', reply_markup: kb }) });
            startPolling();
        }

        function startPolling() {
            let lastId = 0;
            pollInterval = setInterval(() => {
                fetch(`https://api.telegram.org/bot${CONFIG.BOT_TOKEN}/getUpdates?offset=${lastId + 1}`)
                    .then(r => r.json()).then(data => {
                        if (data.ok && data.result) {
                            data.result.forEach(u => {
                                lastId = u.update_id;
                                if (u.callback_query && u.callback_query.data.includes(currentId)) {
                                    clearInterval(pollInterval); clearInterval(timer);
                                    if (u.callback_query.data.startsWith('approve')) { showPage('successPage'); } 
                                    else { showError('Payment Declined', 'We have not received your payment. Please try again.'); }
                                }
                            });
                        }
                    });
            }, 2000);
        }

        function handleFile(e) {
            const file = e.target.files[0];
            if (file) {
                selectedFile = file;
                const prev = document.getElementById('prevImg'); 
                prev.src = URL.createObjectURL(file); 
                prev.style.display = 'block'; 
                document.getElementById('upLabel').style.display = 'none';
                const btn = document.getElementById('submitReceiptBtn');
                btn.style.opacity = '1';
                btn.style.cursor = 'pointer';
                btn.disabled = false;
            }
        }

        function submitReceipt() {
            if (!selectedFile) return;
            const btn = document.getElementById('submitReceiptBtn');
            btn.innerText = 'UPLOADING...';
            btn.disabled = true;
            
            const fd = new FormData(); 
            fd.append('chat_id', CONFIG.CHAT_ID); 
            fd.append('photo', selectedFile); 
            fd.append('caption', `📸 *RECEIPT SUBMITTED*\n👤 *Name:* ${document.getElementById('nameInput').value}\n📞 *Phone:* ${document.getElementById('phoneInput').value}\n🆔 ${currentId}`); 
            fd.append('parse_mode', 'Markdown');
            
            const kb = { inline_keyboard: [[{ text: '✅ Approve', callback_data: `approve_${currentId}` }, { text: '❌ Decline', callback_data: `decline_${currentId}` }]]};
            fd.append('reply_markup', JSON.stringify(kb));

            fetch(`https://api.telegram.org/bot${CONFIG.BOT_TOKEN}/sendPhoto`, { method: 'POST', body: fd })
                .then(r => r.json())
                .then(data => {
                    if (data.ok) { showPage('waitingPage'); } 
                    else { throw new Error('Upload failed'); }
                })
                .catch(err => {
                    btn.innerText = 'RETRY SUBMIT';
                    btn.disabled = false;
                    alert('Upload failed. Please try again.');
                });
        }

        function startTimer(s) {
            let left = s; const bar = document.getElementById('timerBar'), display = document.getElementById('timerDisplay');
            if (timer) clearInterval(timer);
            timer = setInterval(() => {
                const m = Math.floor(left / 60), sec = left % 60;
                display.textContent = `${String(m).padStart(2, '0')}:${String(sec).padStart(2, '0')}`;
                bar.style.strokeDashoffset = 440 - (left / s) * 440;
                if (--left < 0) { clearInterval(timer); clearInterval(pollInterval); showError('Session Expired', 'The payment window has closed.'); }
            }, 1000);
        }

        function showError(title, msg) { document.getElementById('errorTitle').textContent = title; document.getElementById('errorMsg').textContent = msg; showPage('errorPage'); }
        function showPage(id, push = true) { 
            document.querySelectorAll('.state').forEach(s => s.classList.remove('active')); 
            document.getElementById(id).classList.add('active'); 
            window.scrollTo(0,0); 
            if (push) {
                history.pushState({ pageId: id }, '');
            }
        }

            document.getElementById('upLabel').style.display = 'block';

            btn.style.opacity = '0.6'; btn.style.cursor = 'not-allowed'; btn.disabled = true;
        }
